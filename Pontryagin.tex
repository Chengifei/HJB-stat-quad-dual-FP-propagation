Let $\bar{A}_s = (P_s \tilde{\Gamma} - A')$ and $\bar{D}_s = \begin{bmatrix} (M^0)' & P_s L^0\end{bmatrix}$.

Let $\Phi_{t, s}$ denote the state transition matrix that solves
\[
\pdiff{}{s}\Phi_{s, t} = \bar{A}_s \Phi_{s, t}.
\]
Recall (8.5)
\[
\eta(\mu_s) = \begin{bmatrix}0 \\ (L^0)' q^\mu_s\end{bmatrix} + \bar{D}_s' \left[\Phi_{t, s}' x + \int_t^s \Phi_{\tau, s}' (-\tilde{\Gamma} q^\mu_\tau + c_2 L^0 \beta_\tau)\,{\rm d}\tau\right],
\]
where $q^\mu$ solves
\[
\dot{q}^\mu_s = \bar{A}_s q^\mu_s - \bar{D}_s \hat{C} \mu_s, q^\mu_T = 0
\]
and $P$ solves
\[
\dot{P}_s = -S_{c_1} - A' P_s - P_s A + P \tilde{\Gamma} P, PT = 0.
\]
Denote
\[
z^\mu_s = \Phi_{t, s}' x + \int_t^s \Phi_{\tau, s}' (-\tilde{\Gamma} q^\mu_\tau + c_2 L^0 \beta_\tau)\,{\rm d}\tau.
\]
We note that $\Phi_{\tau, s}'$ is the state transition matrix associated with
\[
\pdiff{}{s} \Phi_{\tau, s}' = -\bar{A}_s' \Phi_{\tau, s}'.
\]
Hence $z^\mu, q^\mu$ solves the ODE
\begin{equation}\label{eq:paper_ODE}
\diff{}{s} \begin{bmatrix}z^\mu_s \\ q^\mu_s\end{bmatrix} = \begin{bmatrix}-\bar{A}_s' & -\tilde{\Gamma}\\0_{n \times n} & \bar{A}_s\end{bmatrix} \begin{bmatrix}z^\mu_s \\ q^\mu_s\end{bmatrix} + \begin{bmatrix}0_{n \times k} & c_2 L^0\\\multicolumn{2}{c}{\bar{D}_s}\end{bmatrix} \mu_s, \quad z^\mu_t = x, q^\mu_T = 0,
\end{equation}
where
\[
\eta(\mu_s) = \begin{bmatrix}0 \\ (L^0)' q^\mu_s\end{bmatrix} + \bar{D}_s' z^\mu_s.
\]

We show that (8.5) follows from Pontryagin minimum/staticizing principle.

Consider the staticization problem (3.13)--(3.15).

The Hamiltonian is given by
\[
H(s, \zeta, (\nu, \mu), \lambda) = \lambda^\top (A \zeta + \tilde{B} \nu + c_2 L^0 \beta) + \demi \zeta' C \zeta + \demi \nu' \nu + \tilde{\Theta}(\mu) - \left(\begin{bmatrix}M^0 \zeta \\ 0\end{bmatrix} - \mu\right)' \frac{\hat{C}}{2} \left(\begin{bmatrix}M^0 \zeta \\ 0\end{bmatrix} - \mu\right).
\]
The staticizing controls are given by
\begin{equation}\label{eq:control}
\pdiff{H}{(\nu, \mu)} = 0 \quad \leadsto \quad \nu = -\tilde{B}^\top \lambda, \begin{bmatrix}0 \\ c_2 (L^0)' \lambda\end{bmatrix} - \hat{C} \eta(\mu) + \hat{C} \begin{bmatrix}M^0 \zeta \\ 0\end{bmatrix} = 0.
\end{equation}
The costate equation is given by
\begin{equation}\label{eq:costate}
-\dot{\lambda}^\top = \lambda^\top A + \zeta' C - \begin{bmatrix}M^0 \zeta - \alpha \\ 0\end{bmatrix}' \hat{C} \begin{bmatrix}M^0\\0\end{bmatrix}, \lambda(T) = 0.
\end{equation}
Consider $P_s z_s + q_s$; we supress the superscript $\mu$ for simplicity. Observe that
\begin{align*}
\diff{}{s} (P_s z_s + q_s) &= \dot{P}_s z_s + P_s \dot{z}_s + \dot{q}_s\\
&= (-S_{c_1} - A' P_s - P_s A + P_s \tilde{\Gamma} P_s) z_s + P_s (-\bar{A}_s' z_s - \tilde{\Gamma} q_s + c_2 L^0 \beta_s) + \bar{A}_s q^\mu_s - \bar{D}_s \hat{C} \mu_s.
\end{align*}
Recalling that $\bar{A}_s = P_s \tilde{\Gamma} - A'$, we have
\begin{align*}
\diff{}{s} (P_s z_s + q_s) &= (-S_{c_1} - A' P_s) z_s + P_s (- \tilde{\Gamma} q_s + c_2 L^0 \beta_s) + \bar{A}_s q^\mu_s - \bar{D}_s \hat{C} \mu_s\\
&= (-S_{c_1} - A' P_s) z_s + c_2 P_s L^0 \beta_s - A' q^\mu_s - \bar{D}_s \hat{C} \mu_s
\end{align*}
Recalling $\bar{D}_s = \begin{bmatrix} (M^0)' & P_s L^0\end{bmatrix}$ and $S_{c_1} = C - c_1 (M^0)' M^0$, we have
\begin{align*}
\diff{}{s} (P_s z_s + q_s) &= (c_1 (M^0)' M^0 - C - A' P_s) z_s + c_2 P_s L^0 \beta_s - A' q^\mu_s - c_1 (M^0)' \alpha_s - c_2 P_s L^0 \beta_s\\
&= (c_1 (M^0)' M^0 - C - A' P_s) z_s - A' q^\mu_s - c_1 (M^0)' \alpha_s\\
&= (c_1 (M^0)' M^0 - C) z_s - A' (P_s z_s + q^\mu_s) - c_1 (M^0)' \alpha_s;
\end{align*}
equivalently, we write
\[
-\diff{}{s} (P_s z_s + q_s)' = (P_s z_s + q^\mu_s)' A + z_s' (C - c_1 (M^0)' M^0) + c_1 \alpha_s' M^0
\]
Comparing this to \eqref{eq:costate}, we see that identifying $\zeta \doteq z, \lambda \doteq P z + q$ yields the same equation. \eqref{eq:control} then becomes
\[
\nu = -\tilde{B}^\top (P \zeta + q), \quad \begin{bmatrix}0 \\ c_2 (L^0)' (P \zeta + q)\end{bmatrix} + \hat{C} \eta(\mu) + \hat{C} \begin{bmatrix}M^0 \zeta \\ 0\end{bmatrix} = 0,
\]
where the latter may be rearranged to obtain
\[
\begin{bmatrix}M^0 \zeta \\ (L^0)' P \zeta\end{bmatrix} - \eta(\mu) + \begin{bmatrix}0 \\ (L^0)' q\end{bmatrix} = 0 \quad \leadsto \quad \bar{D}_s' \zeta - \eta(\mu) + \begin{bmatrix}0 \\ (L^0)' q\end{bmatrix} = 0.
\]
Altogether, we obtain \eqref{eq:paper_ODE}.

\begin{equation}\label{eq:LTI_ODE}
\diff{}{s} \begin{bmatrix}z \\ \lambda\end{bmatrix} = \begin{bmatrix}A & -\tilde{\Gamma}\\-S_{c_1} & -A'\end{bmatrix} \begin{bmatrix}z \\ \lambda\end{bmatrix} + \begin{bmatrix} & L^0 \\-(M^0)'\end{bmatrix}\hat{C} \mu, \quad \eta(\mu) = \begin{bmatrix}M^0 z \\ (L^0)' \lambda\end{bmatrix}.
\end{equation}

We now consider the $r$ process, which solves
\begin{align*}
\dot{r}^\mu_s &= (q^\mu_s)' \tilde{\Gamma} q^\mu_s - 2 (q^\mu_s)' L^0 c_2 \beta_s - 2 (\tilde{\Theta}(\mu_t) - \frac{1}{2} \mu_s' \hat{C} \mu_s)\\
&= (q^\mu_s)' \tilde{\Gamma} q^\mu_s - 2 \begin{bmatrix}0 \\ (L^0)' q^\mu_s\end{bmatrix}' \hat{C} \mu_s - 2 \left[\tilde{N}(M^0 z, (L^0)' \lambda) + \frac{1}{2} \left(\mu_s - \begin{bmatrix}M^0 z \\ (L^0)' \lambda\end{bmatrix}\right)' \hat{C} \left(\mu_s - \begin{bmatrix}M^0 z \\ (L^0)' \lambda\end{bmatrix}\right) - \frac{1}{2} \mu_s' \hat{C} \mu_s\right]\\
&= (q^\mu_s)' \tilde{\Gamma} q^\mu_s - 2 \begin{bmatrix}0 \\ (L^0)' q^\mu_s\end{bmatrix}' \hat{C} \mu_s - 2 \left[\tilde{N}(M^0 z, (L^0)' \lambda) - \begin{bmatrix}M^0 z \\ (L^0)' \lambda\end{bmatrix}' \hat{C} \mu_s + \frac{1}{2} \begin{bmatrix}M^0 z \\ (L^0)' \lambda\end{bmatrix}' \hat{C} \begin{bmatrix}M^0 z \\ (L^0)' \lambda\end{bmatrix}\right]\\
&= (q^\mu_s)' \tilde{\Gamma} q^\mu_s + 2 \begin{bmatrix}M^0 z \\ (L^0)' (\lambda - q^\mu_s)\end{bmatrix}' \hat{C} \mu_s - 2 \left(\tilde{N}(M^0 z, (L^0)' \lambda) + \frac{1}{2} \begin{bmatrix}M^0 z \\ (L^0)' \lambda\end{bmatrix}' \hat{C} \begin{bmatrix}M^0 z \\ (L^0)' \lambda\end{bmatrix}\right).
\end{align*}

Let $Z(\cdot, T) = (z_\cdot, \lambda_\cdot)$ denote the process that solves \eqref{eq:LTI_ODE} with boundary condition; that is, we have
\[
\pdiff{Z}{t} = \begin{bmatrix}A & -\tilde{\Gamma}\\-S_{c_1} & -A'\end{bmatrix} Z +
\begin{bmatrix} & L^0 \\-(M^0)'\end{bmatrix} \hat{C} \eta^{-1} \left(\begin{bmatrix}M^0 \\ & (L^0)'\end{bmatrix} Z\right)
\quad \eye_1 Z(0, T) = x, \eye_2 Z(T, T) = 0_{n \times 1}.
\]
Let $Z$ be defined on $[0, \bar{T}]^2$. Then differentiating $Z$ with respect to $T$ yields
\[
\pdiff{{}^2 Z}{T \partial t} = \begin{bmatrix}A & -\tilde{\Gamma}\\-S_{c_1} & -A'\end{bmatrix} \pdiff{Z}{T} +
\begin{bmatrix} & L^0 \\-(M^0)'\end{bmatrix} \hat{C} \nabla [\eta^{-1}] \left(\begin{bmatrix}M^0 \\ & (L^0)'\end{bmatrix} Z\right) \begin{bmatrix}M^0 \\ & (L^0)'\end{bmatrix} \pdiff{Z}{T},
\quad \eye_1 \pdiff{}{T} Z(0_, T) = 0, \eye_2 \left[\pdiff{}{t} Z(T, T) + \pdiff{}{T} Z(T, T)\right] = 0,
\]
or equivalently
\begin{equation}
\pdiff{}{t}\pdiff{Z}{T} = \left(\begin{bmatrix}A & -\tilde{\Gamma}\\-S_{c_1} & -A'\end{bmatrix} +
\begin{bmatrix} & L^0 \\-(M^0)'\end{bmatrix} \hat{C} \nabla [\eta^{-1}] \left(\begin{bmatrix}M^0 \\ & (L^0)'\end{bmatrix} Z\right) \begin{bmatrix}M^0 \\ & (L^0)'\end{bmatrix}\right) \pdiff{Z}{T},
\quad \eye_1 \pdiff{}{T} Z(0_, T) = 0, \eye_2 \left[\pdiff{}{t} Z(T, T) + \pdiff{}{T} Z(T, T)\right] = 0,
\end{equation}
